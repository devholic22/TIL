# L: list, x: value
def linear_search(L, x):
    result = []
    for i in range(len(L)):
        if L[i] == x:
            result.append(i)
    if x not in L:
        result.append(-1)
    return result
```

def binary_search(L, x):
    lower = 0
    upper = len(L) - 1
    while lower <= upper:
        mid = (lower + upper) // 2
        if x < L[mid]:
            upper = mid - 1
        elif x > L[mid]:
            lower = mid + 1
        else:
            return mid
    return -1
```
